# Build a Jobs Scraper

Learn how to scrape job listings from a job board — filter by keyword, extract job details, and build a searchable dataset.

## What You Will Build

A scraper that:
- Searches jobs by keyword and location
- Extracts job title, company, salary, location, and posting date
- Handles paginated results
- Filters by salary range
- Exports to JSON and CSV

## Step 1: Search and List Jobs

<Tabs items={['Python', 'Node.js']}>
  <Tab value="Python">

    ```python
    from crawlingo import Session, Crawl

    session = Session()
    session.rate_limit(2.0)

    crawl = session.crawl("https://jobs.example.com/search?q=engineer&location=remote")
    crawl.follow("a.pagination-next")
    crawl.limit(100)
    crawl.depth(3)
    crawl.field("title", "h2.job-title a")
    crawl.field("company", ".company-name")
    crawl.field("salary", ".salary-range")
    crawl.field("location", ".job-location")
    crawl.field("date", "time.posted-date")

    results = crawl.build()
    ```

  </Tab>
  <Tab value="Node.js">

    ```typescript
    import { Session, Crawl } from 'crawlingo';

    const session = new Session();

    const crawl = new Crawl(
      "https://jobs.example.com/search?q=engineer&location=remote",

    crawl.limit(100);
    crawl.depth(3);
    crawl.field("title", "h2.job-title a");
    crawl.field("company", ".company-name");
    crawl.field("salary", ".salary-range");
    crawl.field("location", ".job-location");
    crawl.field("date", "time.posted-date");

    const results = await crawl.run();
    ```

## Step 2: Clean Salary Data

Parse salary strings into comparable numeric values:

<Tabs items={['Python', 'Node.js']}>
  <Tab value="Python">

    ```python
    import re

    def parse_salary(salary_str: str) -> dict:
        """Parse '$80K - $120K' or '$80,000 - $120,000' into min/max/currency."""
        if not salary_str:
            return {"min": 0, "max": 0, "currency": "USD"}
        nums = re.findall(r'[\d,.]+', salary_str)
        currency = "USD"
        if "£" in salary_str:
            currency = "GBP"
        elif "€" in salary_str:
            currency = "EUR"
        values = []
        for n in nums:
            n = n.replace(",", "")
            if "K" in salary_str.upper():
                values.append(int(float(n) * 1000))
            else:
                values.append(int(float(n)))
        return {
            "min": values[0] if len(values) > 0 else 0,
            "max": values[1] if len(values) > 1 else values[0] if values else 0,
            "currency": currency,
        }
    ```

  </Tab>
  <Tab value="Node.js">

    ```typescript
    function parseSalary(salaryStr: string) {
      if (!salaryStr) return { min: 0, max: 0, currency: "USD" };
      const nums = salaryStr.match(/[\d,.]+/g) || [];
      let currency = "USD";
      if (salaryStr.includes("£")) currency = "GBP";
      else if (salaryStr.includes("€")) currency = "EUR";
      const values = nums.map(n => {
        n = n.replace(/,/g, "");
        if (salaryStr.toUpperCase().includes("K")) {
          return parseInt(n) * 1000;
        }
        return parseInt(n);
      });
      return {
        min: values[0] || 0,
        max: values[1] || values[0] || 0,
        currency,
      };
    }
    ```

## Step 3: Filter by Salary

<Tabs items={['Python', 'Node.js']}>
  <Tab value="Python">

    ```python
    filtered = []
    for r in results:
        d = r.to_dict()
        salary = parse_salary(d.get("salary", ""))
        d["salary_min"] = salary["min"]
        d["salary_max"] = salary["max"]
        d["salary_currency"] = salary["currency"]
        if salary["max"] >= 80000:  # Only jobs paying $80K+
            filtered.append(d)
    print(f"{len(filtered)} jobs paying $80K+ out of {len(results)}")
    ```

  </Tab>
  <Tab value="Node.js">

    ```typescript
    const filtered: Record<string, string | number>[] = [];
    for (const r of results) {
      const d = r.toDict();
      const salary = parseSalary(d.salary ?? "");
      d.salary_min = salary.min;
      d.salary_max = salary.max;
      d.salary_currency = salary.currency;
      if (salary.max >= 80000) {
        filtered.push(d);
      }
    }
    console.log(`${filtered.length} jobs paying $80K+ out of ${results.length}`);
    ```

## Step 4: Fetch Full Job Details

For each filtered job, fetch the full description:

<Tabs items={['Python', 'Node.js']}>
  <Tab value="Python">

    ```python
    from crawlingo import Page

    def fetch_job_details(job_url: str) -> dict:
        page = Page(job_url, session=session, auto_match=True)
        return {
            "description": page.css(".job-description").first().text() if page.css(".job-description").first() else "",
            "skills": [s.text() for s in page.css(".skill-tag")],
            "type": page.css(".job-type").first().text() if page.css(".job-type").first() else "",
        }
    ```

  </Tab>
  <Tab value="Node.js">

    ```typescript
    import { Page } from 'crawlingo';

    async function fetchJobDetails(jobUrl: string) {
      const page = await Page.create(jobUrl, { session, autoMatch: true });
      const description = page.css(".job-description").first()?.text ?? "";
      const skills: string[] = [];
      for (const s of page.css(".skill-tag")) {
        skills.push(s.text);
      }
      const type = page.css(".job-type").first()?.text ?? "";
      return { description, skills, type };
    }
    ```

## Key Takeaways

- Use `Crawl` with `follow` for paginated listing pages
- Post-process extracted data to normalize formats (salary, dates)
- Fetching individual detail pages gives richer data
- Auto-match helps when job boards change their CSS classes
