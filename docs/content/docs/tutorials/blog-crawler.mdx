# Build a Blog Crawler

Learn how to build a blog aggregator that discovers posts via RSS feeds, extracts full content, handles pagination, and deduplicates across sources.

## What You Will Build

A blog aggregator that:
- Parses RSS/Atom feeds from multiple blogs
- Extracts full article content
- Deduplicates by URL
- Exports to CSV and Markdown archives

## Step 1: Parse RSS Feeds

<Tabs items={['Python', 'Node.js']}>
  <Tab value="Python">

    ```python
    import xml.etree.ElementTree as ET
    from crawlingo import Page

    def parse_rss(feed_url: str) -> list[dict]:
        page = Page(feed_url)
        root = ET.fromstring(page.html())

        items = []
        # RSS 2.0
        for item in root.findall(".//item"):
            entry = {
                "title": item.findtext("title", ""),
                "link": item.findtext("link", ""),
                "pubDate": item.findtext("pubDate", ""),
                "description": item.findtext("description", ""),
            }
            items.append(entry)

        # Atom
        for entry in root.findall(".//{http://www.w3.org/2005/Atom}entry"):
            title_el = entry.find("{http://www.w3.org/2005/Atom}title")
            link_el = entry.find("{http://www.w3.org/2005/Atom}link")
            date_el = entry.find("{http://www.w3.org/2005/Atom}published")
            items.append({
                "title": title_el.text if title_el is not None else "",
                "link": link_el.get("href", "") if link_el is not None else "",
                "pubDate": date_el.text if date_el is not None else "",
            })

        return items
    ```

  <Tab value="Node.js">

    ```typescript
    import { Page } from 'crawlingo';
    import { XMLParser } from 'fast-xml-parser';

    const xmlParser = new XMLParser({ ignoreAttributes: false });

    async function parseRSS(feedUrl: string) {
      const page = await Page.create(feedUrl);
      const xml = xmlParser.parse(page.html());
      const items: { title: string; link: string; pubDate: string }[] = [];

      // RSS 2.0
      const rssItems = xml.rss?.channel?.item;
      if (rssItems) {
        const list = Array.isArray(rssItems) ? rssItems : [rssItems];
        for (const item of list) {
          items.push({
            title: item.title ?? "",
            link: item.link ?? "",
            pubDate: item.pubDate ?? "",
          });
        }
      }

      // Atom
      const atomEntries = xml.feed?.entry;
      if (atomEntries) {
        const list = Array.isArray(atomEntries) ? atomEntries : [atomEntries];
        for (const entry of list) {
          items.push({
            title: entry.title ?? "",
            link: entry.link?.['@_href'] ?? entry.link?.href ?? "",
            pubDate: entry.published ?? "",
          });
        }
      }

      return items;
    }
    ```

## Step 2: Discover Posts from Multiple Feeds

<Tabs items={['Python', 'Node.js']}>
  <Tab value="Python">

    ```python
    feeds = [
        "https://blog1.example.com/rss",
        "https://blog2.example.com/feed",
        "https://blog3.example.com/atom",
    ]

    all_posts = []
    seen_urls = set()

    for feed_url in feeds:
        try:
            posts = parse_rss(feed_url)
            for post in posts:
                if post["link"] and post["link"] not in seen_urls:
                    seen_urls.add(post["link"])
                    all_posts.append(post)
            print(f"{feed_url}: {len(posts)} posts")
        except Exception as e:
            print(f"Failed to parse {feed_url}: {e}")

    print(f"Total unique posts: {len(all_posts)}")
    ```

  </Tab>
  <Tab value="Node.js">

    ```typescript
    const feeds = [
      "https://blog1.example.com/rss",
      "https://blog2.example.com/feed",
    ];

    const allPosts: { title: string; link: string; pubDate: string }[] = [];
    const seenUrls = new Set<string>();

    for (const feedUrl of feeds) {
      try {
        const posts = await parseRSS(feedUrl);
        for (const post of posts) {
          if (post.link && !seenUrls.has(post.link)) {
            seenUrls.add(post.link);
            allPosts.push(post);
          }
        }
        console.log(`${feedUrl}: ${posts.length} posts`);
      } catch (e) {
        console.log(`Failed to parse ${feedUrl}: ${e}`);
      }
    }
    console.log(`Total unique posts: ${allPosts.length}`);
    ```

## Step 3: Extract Full Article Content

<Tabs items={['Python', 'Node.js']}>
  <Tab value="Python">

    ```python
    from crawlingo import Page

    def extract_article(url: str) -> dict:
        page = Page(url, auto_match=True)
        article = page.css("article, .post, .entry, .content")
        content = article.first().text() if article.first() else page.css("body").first().text()
        return {
            "title": page.title(),
            "content": content[:5000],
            "author": page.css(".author").first().text() if page.css(".author").first() else "",
        }
    ```

  </Tab>
  <Tab value="Node.js">

    ```typescript
    import { Page } from 'crawlingo';

    async function extractArticle(url: string) {
      const page = await Page.create(url, { autoMatch: true });
      const article = page.css("article, .post, .entry, .content");
      const content = article.first()?.text ?? page.css("body").first()?.text ?? "";
      const author = page.css(".author").first()?.text ?? "";
      return { title: page.title(), content: content.slice(0, 5000), author };
    }
    ```

## Step 4: Generate Blog Archive

    ```python
    with open("blog_archive.csv", "w", newline="") as f:
    ```

## Key Takeaways

- RSS/Atom feeds provide structured metadata before fetching pages
- Deduplicate by URL when aggregating multiple sources
- Use broad CSS selectors for article content across different blog themes
- Auto-match helps when individual blogs have unique layouts
- Extract content in batches to manage memory and rate limits
