# Build a Price Tracker

Learn how to use Crawlingo's `Watch` feature to monitor product prices and receive callbacks when they change.

## What You Will Build

A price monitor that:
- Polls a product page every 5 minutes
- Extracts current price, stock status, and product name
- Fires callbacks on price drops, stock changes, or any change
- Logs all changes to a JSON file for historical analysis

## Prerequisites

- Crawlingo installed
- A product URL to monitor

## Step 1: Set Up the Watch

## Step 2: Register Callbacks

<Tabs items={['Python', 'Node.js']}>
  <Tab value="Python">

    ```python
    import json
    from datetime import datetime

    price_history = []

    def on_price_drop(event):
        print(f"💰 PRICE DROP! {event.field}: {event.old_value} -> {event.new_value}")
        price_history.append({
            "timestamp": datetime.now().isoformat(),
            "field": event.field,
            "old": event.old_value,
            "new": event.new_value,
            "type": event.change_type,
        })

    def on_stock_change(event):
        print(f"📦 STOCK CHANGE: {event.new_value}")
        price_history.append({
            "timestamp": datetime.now().isoformat(),
            "field": event.field,
            "old": event.old_value,
            "new": event.new_value,
            "type": event.change_type,
        })

    def on_any_change(event):
        print(f"[{event.detected_at}] {event.field}: {event.change_type}")
        price_history.append({
            "timestamp": event.detected_at,
            "field": event.field,
            "old": event.old_value,
            "new": event.new_value,
            "type": event.change_type,
        })

    watch.on_price_change(on_price_drop)
    watch.on_stock_change(on_stock_change)
    watch.on_change(on_any_change)
    ```

  </Tab>
  <Tab value="Node.js">

    ```typescript
    import { WatchChangeEvent } from 'crawlingo';

    interface HistoryEntry {
      timestamp: string;
      field: string;
      old?: string;
      new?: string;
      type: string;
    }

    const priceHistory: HistoryEntry[] = [];

    watch.run((err: Error | null, event: WatchChangeEvent) => {
      if (err) {
        console.error("Watch error:", err);
        return;
      }
      console.log(
        `[${new Date().toISOString()}] ${event.field}: ${event.changeType}`
      );
      if (event.changeType === "PriceChange") {
        console.log(`💰 Price: ${event.oldValue} -> ${event.newValue}`);
      }
      if (event.changeType === "StockChange") {
        console.log(`📦 Stock: ${event.newValue}`);
      }
      priceHistory.push({
        timestamp: new Date().toISOString(),
        field: event.field,
        old: event.oldValue,
        new: event.newValue,
        type: event.changeType,
      });
    });
    ```

## Step 3: Save History Periodically

<Tabs items={['Python', 'Node.js']}>
  <Tab value="Python">

    ```python
    import threading
    import time

    def save_history():
        while True:
            time.sleep(3600)  # Save every hour
            with open("price_history.json", "w") as f:
                json.dump(price_history, f, indent=2)
            print(f"Saved {len(price_history)} events")

    save_thread = threading.Thread(target=save_history, daemon=True)
    save_thread.start()

    watch.run()
    ```

  </Tab>
  <Tab value="Node.js">

    ```typescript
    import { writeFile } from 'fs/promises';

    setInterval(async () => {

        "price_history.json",
        JSON.stringify(priceHistory, null, 2)
      );
      console.log(`Saved ${priceHistory.length} events`);
    }, 3600000); // Every hour
    ```

## Step 5: Stop Gracefully

<Tabs items={['Python', 'Node.js']}>

    def handle_signal(sig, frame):

## Complete Price Tracker

<Tabs items={['Python', 'Node.js']}>
  <Tab value="Python">

    ```python
    from crawlingo import Session, Watch
    import json, signal, threading, time

    session = Session()
    session.rate_limit(6.0)

    watch = Watch("https://shop.example.com/product/42", session)
    watch.field("name", "h1.product-title")
    watch.field("price", "span.price")
    watch.field("stock", ".availability")
    watch.interval(300)

    history = []

    watch.on_price_change(lambda ev: history.append({"ts": ev.detected_at, "field": ev.field, "old": ev.old_value, "new": ev.new_value, "type": ev.change_type}))
    watch.on_stock_change(lambda ev: history.append({"ts": ev.detected_at, "field": ev.field, "old": ev.old_value, "new": ev.new_value, "type": ev.change_type}))
    watch.on_change(lambda ev: print(f"[{ev.detected_at}] {ev.field}: {ev.change_type}"))

    def save():
        while True:
            time.sleep(300)
            with open("price_history.json", "w") as f:
                json.dump(history, f, indent=2)

    threading.Thread(target=save, daemon=True).start()
    signal.signal(signal.SIGINT, lambda s, f: (watch.stop(), exit(0)))
    watch.run()
    ```

  </Tab>
  <Tab value="Node.js">

    ```typescript
    import { Session, Watch, WatchChangeEvent } from 'crawlingo';
    import { writeFile } from 'fs/promises';

    const session = new Session();
    session.rateLimit(6.0);

    const watch = new Watch("https://shop.example.com/product/42", session);
    watch.field("name", "h1.product-title");
    watch.field("price", "span.price");
    watch.field("stock", ".availability");
    watch.interval(300);

    const history: Record<string, string | undefined>[] = [];

    watch.run((err, event) => {
      if (err) return console.error(err);
      console.log(`[${new Date().toISOString()}] ${event.field}: ${event.changeType}`);
      history.push({
        ts: new Date().toISOString(),
        field: event.field,
        old: event.oldValue,
        new: event.newValue,
        type: event.changeType,
      });
    });

    setInterval(async () => {
      await writeFile("price_history.json", JSON.stringify(history, null, 2));
    }, 300000);

    process.on('SIGINT', () => { watch.stop(); process.exit(0); });
    ```

  </Tab>
</Tabs>

## Key Takeaways

- `Watch` is purpose-built for periodic monitoring
- Separate callbacks for each change type (price, stock, added, removed)
- Always save history periodically to avoid data loss
- Use `run_async()` or threads for non-blocking operation
- Stop watches gracefully on shutdown
